    <script>
        // Load confessions from localStorage or initialize empty array
        let confessions = JSON.parse(localStorage.getItem('confessions')) || [];
        
        // DOM elements
        const confessionForm = document.getElementById('confessionForm');
        const confessionWall = document.getElementById('confessionWall');
        const emptyWall = document.getElementById('emptyWall');
        
        // Display existing confessions
        function displayConfessions() {
            confessionWall.innerHTML = '';
            
            if (confessions.length === 0) {
                emptyWall.style.display = 'block';
                confessionWall.appendChild(emptyWall);
                return;
            }
            
            emptyWall.style.display = 'none';
            
            // Sort confessions by date (newest first)
            confessions.sort((a, b) => new Date(b.date) - new Date(a.date));
            
            confessions.forEach(confession => {
                const confessionCard = document.createElement('div');
                confessionCard.className = 'confession-card';
                
                const confessionText = document.createElement('p');
                confessionText.className = 'confession-text';
                confessionText.textContent = confession.message;
                
                const confessionAuthor = document.createElement('p');
                confessionAuthor.className = 'confession-author';
                confessionAuthor.textContent = `- ${confession.name}`;
                
                const confessionDate = document.createElement('p');
                confessionDate.className = 'confession-date';
                confessionDate.textContent = new Date(confession.date).toLocaleDateString();
                
                confessionCard.appendChild(confessionText);
                confessionCard.appendChild(confessionAuthor);
                confessionCard.appendChild(confessionDate);
                
                confessionWall.appendChild(confessionCard);
            });
        }
        
        // Handle form submission
        confessionForm.addEventListener('submit', function(e) {
            e.preventDefault();
            
            const nameInput = document.getElementById('secretName');
            const messageInput = document.getElementById('confessionMessage');
            
            const name = nameInput.value.trim() || 'Anonymous';
            const message = messageInput.value.trim();
            
            if (!message) {
                alert('Please write a confession before submitting.');
                return;
            }
            
            // Create new confession
            const newConfession = {
                name: name,
                message: message,
                date: new Date().toISOString()
            };
            
            // Add to array and save to localStorage
            confessions.push(newConfession);
            localStorage.setItem('confessions', JSON.stringify(confessions));
            
            // Reset form
            confessionForm.reset();
            
            // Update display
            displayConfessions();
            
            // Scroll to the new confession
            window.scrollTo({
                top: document.querySelector('.confession-wall').offsetTop,
                behavior: 'smooth'
            });
        });
        
        // Initial display
        displayConfessions();
    </script>
